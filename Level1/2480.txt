#include <iostream>
using namespace std;

int main() {
    int arr[7] = {0};
    int x;

    for (int i = 0; i < 3; i++) {
        cin >> x;
        arr[x]++;
    }

    int result = 0;
    for (int i = 1; i < 7; i++) {
        if (arr[i] == 3) {
            result = 10000 + i * 1000;
            break;
        }
        if (arr[i] == 2) {
            result = 1000 + i * 100;
            break;
        }
    }

    if (result == 0) {
        for (int i = 6; i >= 1; i--) {
            if (arr[i] == 1) {
                result = i * 100;
                break;
            }
        }
    }

    cout << result;

    return 0;
}
